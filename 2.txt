#include <stdio.h>
#include <stdlib.h>
#include <time.h>
int main()
{
    clock_t start, stop;
    float tim;
    srand(time(0));
	int r1,c1,r2,c2;
    printf("\nEnter number of rows of the first matrix: ");
    scanf ("%d",&r1);
    printf("\nEnter number of columns of the first matrix: ");
    scanf ("%d",&c1);
    printf("\nEnter number of rows of the second matrix: ");
    scanf ("%d",&r2);
    printf("\nEnter number of columns of the second matrix: ");
    scanf ("%d",&c2);
    
    if(r1==r2 && c1==c2)
    {
     int m1[r1][c1];
     int m2[r2][c2];
     int result[r1][c1];
     
     //printf("\nEnter first matrix\n");
     for (int i = 0 ; i < r1; i++)
     {
         for (int j = 0; j < c1 ; j++)
         {
             m1[i][j] = rand() % 100;
         }
     }
     //printf("\nEnter second matrix\n");
     for (int i = 0 ; i < r2; i++)
     {
         for (int j = 0; j < c2 ; j++)
         {
             m2[i][j] = rand() % 100;
         }
     }
     start=clock();
     for (int i = 0 ; i < r2; i++)
     {
         for (int j = 0; j < c2 ; j++)
         {
             result[i][j] = m1[i][j] + m2[i][j];
         }
     }
     
     printf("\nResultant matrix:\n");
     for (int i = 0 ; i < r2; i++)
     {
         for (int j = 0; j < c2 ; j++)
         {
             printf("\t%d",result[i][j]);
         }
         printf("\n");
     }
     stop=clock();
     tim = (float)(stop-start)/CLOCKS_PER_SEC;
     printf("Addition time: %0.50f\n",tim);
        
        
        
        
        
    }
    else
        printf("\nAddition can not be done\n");

    return 0;
}
